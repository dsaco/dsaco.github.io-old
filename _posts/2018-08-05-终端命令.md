{% highlight shell %}
# 后台启动
nohup npm run xxx  &
# 查看端口 search(node)
ps -aux |grep node  
# 远程拷贝文件
$ scp 214103956280095.pem root@45.77.29.14:nginx/ssl
# 从远端拷贝文件
scp -r root@45.77.29.14:/etc /whatever
# 解决 REMOTE HOST IDENTIFICATION HAS CHANGED!
ssh-keygen -R + 输入服务器的IP
# 返回当前目录
pwd
# mac 查看端口占用情况
lsof -i tcp:8000
{% endhighlight %}
# 赋予权限
sudo chown username /etc/whatever
